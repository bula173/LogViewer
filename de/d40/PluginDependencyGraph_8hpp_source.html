219 bool HasCycle(const std::string& id, std::set<std::string>& visited,
220 std::set<std::string>& recursionStack) const
221 {
222 if (recursionStack.count(id)) {
223 util::Logger::Warn("Circular dependency involving: {}", id);
224 return true;
225 }
226 if (visited.count(id)) {
227 return false;
228 }
229
230 visited.insert(id);
231 recursionStack.insert(id);
232
233 auto it = m_nodes.find(id);
234 if (it != m_nodes.end()) {
235 for (const auto& dep : it->second.dependencies) {
236 if (HasCycle(dep, visited, recursionStack)) {
237 return true;
238 }
239 }
240 }
241
242 recursionStack.erase(id);
243 return false;
244 }

249 void TopologicalSort(const std::string& id,

260 for (const auto& dep : it->second.dependencies) {
261 TopologicalSort(dep, visited, order);
262 }
263 }
