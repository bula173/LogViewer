282 // If plugins are already enabled, push the callbacks immediately.
283 // This avoids ordering issues (e.g. host sets callbacks after plugin enable).
284 if (!m_hostUiOpaque || m_hostUiCallbacks.size < sizeof(PluginHostUiCallbacks))
285 return;
286
287 using SetHostUiCallbacksFn = void(*)(void*, void*, const PluginHostUiCallbacks*);
288 for (auto& [id, info] : m_plugins)
289 {
290 (void)id;
291 if (!info.enabled) continue;
292 if (!info.pluginOpaqueHandle) continue;
293 if (!info.pluginSetHostUiCallbacks) continue;
294 auto fn = reinterpret_cast<SetHostUiCallbacksFn>(info.pluginSetHostUiCallbacks);
