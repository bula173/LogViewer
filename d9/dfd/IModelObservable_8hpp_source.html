101 void UnregisterView(const IView* view)
102 {
103 if (!view) {
104 return;
105 }
106
107 ViewPtr found;
108 {
109 std::unique_lock lock(m_viewsMutex);
110
111 for (auto& weak : m_weakViews) {
112 if (auto v = weak.lock(); v && v.get() == view) {
113 found = v;
114 break;
115 }
116 }
117
118 m_weakViews.erase(
119 std::remove_if(m_weakViews.begin(), m_weakViews.end(),
120 [view](const auto& weak) {
121 auto shared = weak.lock();
122 return !shared || shared.get() == view;
123 }),
124 m_weakViews.end());
125
126 util::Logger::Trace("View unregistered, remaining: {} views",
127 m_weakViews.size());
128 }
129 // Call into view AFTER releasing the lock
130 if (found) {
131 found->OnDetachedFromModel();
132 }
133 }

162 // This prevents deadlock from lock upgrade (shared→unique) and
163 // re-entrancy if a view calls back into the model during notification.
