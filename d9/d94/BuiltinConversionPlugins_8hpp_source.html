28 {
29 if (value.empty() || value.length() % 2 != 0)
30 return std::string(value);
31
32 // Check if all characters are hex digits
33 bool isHex = std::all_of(value.begin(), value.end(),
34 [](char c) { return std::isxdigit(static_cast<unsigned char>(c)); });
35
36 if (!isHex)

38
39 std::string result;
40 result.reserve(value.length() / 2);

44 std::string byteStr(value.substr(i, 2));
45 try
46 {
47 int byteValue = std::stoi(byteStr, nullptr, 16);
48 result += static_cast<char>(byteValue);
49 }
50 catch (...)
51 {
52 return std::string(value); // Return original on error
53 }
54 }
55
56 return result;
57 }
