75 explicit SpdLogger(LogLevel level = LogLevel::Debug, const std::string& logFile = "", const std::string& name = "LogViewer")
76 {
77 if (m_logger.get() == nullptr) {
78 std::vector<spdlog::sink_ptr> sinks;
79
80
81 // Console sink for debug output
82 auto console_sink = std::make_shared<spdlog::sinks::stdout_color_sink_mt>();
83 console_sink->set_level(toSpdlogLevel(level));
84 sinks.push_back(console_sink);
85
86 // Additional spdlog configuration can go here
87 if (!logFile.empty()) {
88 auto file_sink = std::make_shared<spdlog::sinks::basic_file_sink_mt>(logFile, true);
89 file_sink->set_level(toSpdlogLevel(level));
90 sinks.push_back(file_sink);
91 }
92
93 auto logger = std::make_shared<spdlog::logger>(name, sinks.begin(), sinks.end());
94 logger->set_level(toSpdlogLevel(level));
95 spdlog::set_default_logger(logger);
96 m_logger = std::move(logger);
97 // Flush on every log call (trace and above covers all levels).
98 // Do NOT use spdlog::flush_every(): it spawns a background thread
99 // that sleeps periodically and then flushes to disk — the exact
100 // pattern (background thread + timed sleep + file I/O) that
101 // triggers Windows Defender's ransomware ML heuristic (Wacatac.C!ml).
102 spdlog::flush_on(spdlog::level::trace);
103 }
104 }
