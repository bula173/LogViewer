174 m_observers.end());

189 {
190 std::shared_lock lock(m_observersMutex);

192 // Log event for debugging
193 LogEvent(event);

195 // Notify all live observers
196 for (auto& weak : m_observers) {

201 util::Logger::Error("Observer exception: {}", e.what());

204 }

276 void CleanupDeadObserversUnlocked()
